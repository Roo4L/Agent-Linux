    #!/usr/bin/env bats
    # tests/bats/10-installer.bats — INST-01, INST-02, INST-05, DOC-02.

      # Snapshot /etc + /home/agent before; re-run; diff MUST be (mostly) empty.
      local pre=/tmp/agentlinux-pre-$$ post=/tmp/agentlinux-post-$$

      if ! diff -q "$pre" "$post" >/dev/null; then

    }
