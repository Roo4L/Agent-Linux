    Final file shape (target ~100 lines):

    ```bash
    #!/usr/bin/env bash
    # plugin/bin/agentlinux-install — AgentLinux installer entrypoint
    # Phase 2 (v0.3.0): provisions the agent user + PATH wiring across six modes.
    # Refs: docs/HARNESS.md §1.1, .claude/skills/agentlinux-installer/SKILL.md, .planning/phases/02-installer-foundation-agent-user/02-RESEARCH.md §Pattern 1, Example 1
    set -euo pipefail

    readonly AGENTLINUX_VERSION="0.3.0"
    readonly LOG_FILE="${AGENTLINUX_LOG:-/var/log/agentlinux-install.log}"
    readonly BIN_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
    readonly LIB_DIR="$(cd "$BIN_DIR/../lib" && pwd)"
    readonly PROV_DIR="$(cd "$BIN_DIR/../provisioner" && pwd)"

    # Initialize the log file BEFORE redirecting stdio into it.
    # `install -m 0644 /dev/null` creates or truncates the file with mode 0644 atomically.
    # The log is root-owned (installer runs as root); INST-05 greps it for EACCES|permission denied.
    # Note: `install` follows symlinks — acceptable on a clean root installer; documented in A7 (RESEARCH Assumptions Log).
    install -m 0644 /dev/null "$LOG_FILE" 2>/dev/null || {
      printf 'agentlinux-install: cannot create %s (are you root?)\n' "$LOG_FILE" >&2
      exit 64
    }

    # Tee everything to the log. Merge stderr into stdout (`2>&1`) AFTER tee so both
    # streams land in the single greppable transcript. Pitfall 6: `trap wait EXIT`
    # ensures the async `tee` child flushes before the shell exits.
    exec > >(tee -a "$LOG_FILE") 2>&1
    trap 'wait' EXIT

    # shellcheck source=../lib/log.sh
    . "$LIB_DIR/log.sh"

    on_error() {
      local exit_code=$?
      local line_no=${BASH_LINENO[0]:-unknown}
      local src=${BASH_SOURCE[1]:-unknown}
      log_error "installer failed at $(basename "${src}"):${line_no} (exit ${exit_code})"
      log_error "full transcript: ${LOG_FILE}"
      # Flush the tee child before exit.
      sync 2>/dev/null || true
      exit "$exit_code"
    }
    trap on_error ERR

    # shellcheck source=../lib/distro_detect.sh
    . "$LIB_DIR/distro_detect.sh"
    # shellcheck source=../lib/idempotency.sh
    . "$LIB_DIR/idempotency.sh"
    # shellcheck source=../lib/as_user.sh
    . "$LIB_DIR/as_user.sh"

    usage() {
      cat <<EOF
    agentlinux-install v${AGENTLINUX_VERSION} — provision an agent user and environment

    Usage: agentlinux-install [OPTIONS]

    Options:
      -h, --help       Show this help and exit
      -V, --version    Print version and exit
          --verbose    Enable DEBUG-level logging
          --purge      (Phase 4 stub) remove the agent user and installer state

    The installer must run as root. Supported hosts: Ubuntu 22.04 and 24.04.
    Transcript: ${LOG_FILE}
    EOF
    }

    parse_args() {
      while [[ $# -gt 0 ]]; do
        case "$1" in
          -h | --help)
            usage
            exit 0
            ;;
          -V | --version)
            printf '%s\n' "$AGENTLINUX_VERSION"
            exit 0
            ;;
          --verbose)
            export AGENTLINUX_LOG_LEVEL=DEBUG
            log_debug "verbose logging enabled"
            ;;
          --purge)
            log_warn "--purge is a Phase 4 stub; no action taken in v0.3.0 Phase 2"
            exit 0
            ;;
          *)
            log_error "unknown argument: $1"
            usage >&2
            exit 64
            ;;
        esac
        shift
      done
    }

    require_root() {
      if [[ ${EUID:-$(id -u)} -ne 0 ]]; then
        log_error "agentlinux-install must run as root (EUID != 0). Re-run under sudo."
        exit 64
      fi
    }

    run_provisioners() {
      shopt -s nullglob
      local steps=("$PROV_DIR"/[0-9][0-9]-*.sh)
      shopt -u nullglob
      if [[ ${#steps[@]} -eq 0 ]]; then
        log_warn "no provisioner scripts found under ${PROV_DIR} (Phase 2 provisioners land in sibling plans)"
        return 0
      fi
      for step in "${steps[@]}"; do
        log_info "running $(basename "$step")"
        # shellcheck disable=SC1090
        . "$step"
      done
    }

    main() {
      parse_args "$@"
      require_root
      log_info "agentlinux-install v${AGENTLINUX_VERSION} starting"
      detect_distro
      run_provisioners
      log_info "agentlinux-install complete (transcript: ${LOG_FILE})"
    }

    main "$@"
    ```

    Key invariants:
    - Log file is created via `install -m 0644 /dev/null` BEFORE the tee redirect (avoids tee racing on a missing file) and BEFORE sourcing log.sh (so a failure here can still print to stderr directly).
    - Handling when not root: the `install -m` line succeeds under a non-root UID because `/var/log/agentlinux-install.log` may already exist OR be writable; real root check is in `require_root` called by `main`. Practical note: on GH Actions test runners + Docker, the installer always runs as root; the `exit 64` in the fallback is defense in depth.
    - `parse_args` runs BEFORE `require_root` so `--help` / `--version` work without sudo (UX contract).
    - `run_provisioners` tolerates zero provisioners — Plans 03 and 04 can land provisioners independently without breaking this plan's verification step.
    - `shopt -s nullglob` is scoped; restored with `shopt -u nullglob` after the glob expansion.
    - All sources to lib files have `# shellcheck source=../lib/<file>` hints so shellcheck `--external-sources` can follow.
    - `#!/usr/bin/env bash` preserved from Phase 1 stub; Phase 2 body replaces everything after it.
