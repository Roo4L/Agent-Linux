```typescript
import Ajv2020 from 'ajv/dist/2020.js';
import addFormats from 'ajv-formats';
import { readFile } from 'node:fs/promises';
import { fileURLToPath } from 'node:url';
import { dirname, join } from 'node:path';

const HERE = dirname(fileURLToPath(import.meta.url));
const SCHEMA_PATH = join(HERE, '..', '..', '..', 'catalog', 'schema.json');

let cached: any = null;
export async function getValidator() {
  if (cached) return cached;
  const schema = JSON.parse(await readFile(SCHEMA_PATH, 'utf8'));
  const ajv = new Ajv2020({ allErrors: true, strict: true });
  addFormats(ajv);
  cached = ajv.compile(schema);
  return cached;
}

export function formatErrors(errors: any[] | null | undefined): string {
  if (!errors || errors.length === 0) return '(no errors)';
  return errors
    .map((e) => `  • ${e.instancePath || '(root)'}: ${e.message} ${JSON.stringify(e.params)}`)
    .join('\n');
}
```

The `ajv/dist/2020.js` (explicit `.js` extension) is load-bearing per Pitfall 1 + tsconfig moduleResolution=NodeNext. Leave the `.js` extension there even though it feels redundant.

```typescript
import { execFile } from 'node:child_process';
import { promisify } from 'node:util';

const pexecFile = promisify(execFile);

export async function asUser(
  user: string,
  argv: string[],
  opts: { env: Record<string, string>; cwd?: string; timeout?: number }
): Promise<{ exitCode: number; stdout: string; stderr: string }> {
  try {
    const { stdout, stderr } = await pexecFile(
      'sudo',
      ['-u', user, '-H', '-E', '--', ...argv],
      { env: opts.env, cwd: opts.cwd, timeout: opts.timeout, maxBuffer: 10 * 1024 * 1024 }
    );
    return { exitCode: 0, stdout, stderr };
  } catch (err: any) {
    if (typeof err.code === 'number' || typeof err.code === 'string') {
      // execFile promise rejects with {code, stdout, stderr} on non-zero exit
      return { exitCode: Number(err.code) || 1, stdout: err.stdout ?? '', stderr: err.stderr ?? '' };
    }
    throw err;
  }
}
```

Three load-bearing flags — must match plugin/lib/as_user.sh:
- `-H` force HOME
- `-E` preserve env (subject to secure_path in sudoers; Pitfall 5)
- `--` end option parsing

Use `execFile` (array argv), NOT `exec` (shell-string) — prevents injection via catalog-entry `id` values. Per Anti-Pattern "Shell injection in <name> arg."
