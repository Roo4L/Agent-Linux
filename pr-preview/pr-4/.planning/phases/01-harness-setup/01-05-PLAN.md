**A. `tests/harness/run.sh`** — single entry point. Executable. Must be idempotent, exit 0 iff the suite passes.
```bash
#!/usr/bin/env bash
# tests/harness/run.sh
# Runs the Phase 1 harness acceptance suite.
# Exit 0 iff every HRN-XX, TST-06, TST-07 artifact required by ROADMAP.md §Phase 1 is present.
set -euo pipefail

HERE="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
cd "$(cd "$HERE/../.." && pwd)"

if ! command -v bats >/dev/null 2>&1; then
  cat <<'EOF' >&2
tests/harness/run.sh: bats is not installed on PATH.

Install bats-core:
  Ubuntu:  sudo apt install bats
  macOS:   brew install bats-core
  npm:     npm install -g bats
  OR run via docker: docker run --rm -v "$PWD":/code -w /code bats/bats:latest tests/harness/
EOF
  exit 127
fi

exec bats "$HERE"/*.bats
```

```bash
# tests/harness/run.sh
# Phase 1 acceptance suite. Exit 0 iff every HRN-XX, TST-06, TST-07 deliverable
# from ROADMAP.md §Phase 1 is present.
set -euo pipefail

HERE="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
REPO_ROOT="$(cd "$HERE/../.." && pwd)"
cd "$REPO_ROOT"

echo "== harness: bats suite =="
if ! command -v bats >/dev/null 2>&1; then

tests/harness/run.sh: bats is not installed on PATH.

Install bats-core:
  Ubuntu:  sudo apt install bats
  macOS:   brew install bats-core
  npm:     npm install -g bats
  OR run via docker: docker run --rm -v "$PWD":/code -w /code bats/bats:latest tests/harness/
EOF
  exit 127
fi

bats "$HERE"/*.bats
BATS_STATUS=$?

echo
echo "== harness: pre-commit smoke (optional) =="
if command -v pre-commit >/dev/null 2>&1; then
  # pre-commit must run green on the Phase 1 deliverable. If this fails, Phase 1 is not done.
  pre-commit run --all-files --show-diff-on-failure
  echo "pre-commit: OK"
else
  echo "pre-commit not installed on PATH; skipping smoke. CI installs it in test.yml." >&2
fi

exit "$BATS_STATUS"
```
