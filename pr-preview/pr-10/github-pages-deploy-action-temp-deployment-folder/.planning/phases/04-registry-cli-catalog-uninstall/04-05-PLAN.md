<objective>
Implement `agentlinux pin <name>=<target>` per CLI-07 + ADR-011 sticky-override semantics:
- `=curated` clears the sticky flag (source='curated', sticky=false). The entry returns to following the catalog's curated pin on subsequent upgrades.
- `=latest` sets sticky=true + source='latest'. The entry surfaces in `list` / `upgrade` but is SKIPPED by `upgrade --all-latest` (already at intent) and SKIPPED by `upgrade --reset-all-curated` UNLESS the flag is explicitly passed (ADR-011: power users aren't re-nagged).
- `=2.1.7` (exact semver) sets sticky=true + source='pinned' + version=2.1.7. The entry is pinned at this specific version; `upgrade` never touches it until another `pin` runs.

Pin is a STATE MUTATION on the sentinel. It does NOT re-invoke install.sh — the user is asserting intent about what the existing install means. If the sentinel does not exist (agent not installed), pin fails with a clear message (intent without install is meaningless for v0.3.0; Phase 5+ may allow "pre-declared intent" but not v0.3.0).

Purpose: CLI-07 + the sticky-override UX from ADR-011 + Homebrew `brew pin` precedent.

Output: Working `agentlinux pin <spec>` verb with three target types; rejection paths for malformed spec / missing sentinel / unknown agent; unit tests covering each.

Step 1 — Create plugin/cli/src/commands/pin.ts:
```typescript
// plugin/cli/src/commands/pin.ts — CLI-07 sticky-override semantics per ADR-011.
//
// pin is a STATE-ONLY verb: it updates /opt/agentlinux/state/installed.d/<id>.json
// to record the user's intent (curated | latest | exact version). It does NOT
// re-invoke install.sh. Upgrade reads the sticky flag and skips sticky entries
// unless --reset-all-curated. This preserves the Homebrew brew-pin UX: power
// users running ahead of the curated set are not re-nagged each release.
//
// Design Q (RESEARCH Open Q4): pin=latest resolves AT UPGRADE TIME, not at pin
// time. Between pin and next upgrade, the sentinel.version is the last known
// installed version; list surfaces any divergence.

import semver from 'semver';
import { loadCatalog } from '../catalog/loader.js';
import { readSentinel, writeSentinel } from '../state/sentinel.js';
import type { Sentinel } from '../types.js';

export interface PinOpts { /* no current flags */ json?: boolean; }

export type PinTarget =
  | { name: string; target: 'curated' }
  | { name: string; target: 'latest' }
  | { name: string; target: 'version'; version: string };

export function parsePinSpec(spec: string): PinTarget {
  const eq = spec.indexOf('=');
  if (eq <= 0) {
    throw new Error(
      `agentlinux pin: expected '<name>=<target>' (got '${spec}');\n` +
      `  valid targets: curated, latest, or exact semver like 2.1.7`
    );
  }
  const name = spec.slice(0, eq);
  const tgt = spec.slice(eq + 1);
  if (!name) {
    throw new Error(`agentlinux pin: empty name in '${spec}'`);
  }
  if (tgt === 'curated') return { name, target: 'curated' };
  if (tgt === 'latest')  return { name, target: 'latest' };
  if (semver.valid(tgt)) return { name, target: 'version', version: tgt };
  throw new Error(
    `agentlinux pin: invalid target '${tgt}' in '${spec}';\n` +
    `  valid targets: curated, latest, or exact semver (e.g. 2.1.7)`
  );
}

export async function pinCmd(spec: string, _opts: PinOpts = {}): Promise<void> {
  let parsed: PinTarget;
  try {
    parsed = parsePinSpec(spec);
  } catch (err: any) {
    console.error(err.message);
    process.exit(64);
  }

  const catalog = await loadCatalog({ validate: true });
  const entry = catalog.agents.find((a) => a.id === parsed.name);
  if (!entry) {
    console.error(`agentlinux: no such agent in catalog: ${parsed.name}`);
    console.error(`  available: ${catalog.agents.filter((a) => !a.test_only).map((a) => a.id).join(', ')}`);
    process.exit(64);
  }

  const existing = await readSentinel(entry.id);
  if (!existing) {
    console.error(`agentlinux: ${entry.id} is not installed — run 'agentlinux install ${entry.id}' first`);
    process.exit(1);
  }

  // Compute next sentinel state.
  let next: Sentinel;
  switch (parsed.target) {
    case 'curated':
      // Clear the sticky flag; re-align to catalog. Version stays at whatever
      // is currently installed (sentinel.version); a subsequent upgrade will
      // classify the entry and re-install if divergent.
      next = { ...existing, sticky: false, source: 'curated' };
      console.log(`${entry.id}: pin cleared (source=curated, sticky=false)`);
      break;
    case 'latest':
      next = { ...existing, sticky: true, source: 'latest' };
      console.log(`${entry.id}: pinned to follow upstream latest (sticky=true); next 'upgrade --all-latest' will resolve`);
      break;
    case 'version':
      next = { ...existing, sticky: true, source: 'pinned', version: parsed.version };
      console.log(`${entry.id}: pinned to ${parsed.version} (sticky=true)`);
      break;
  }
  await writeSentinel(next);
}
```

```typescript
import { test, describe, before, beforeEach, after } from 'node:test';
import assert from 'node:assert/strict';
import { mkdtemp, rm, writeFile, mkdir } from 'node:fs/promises';
import { join } from 'node:path';
import { tmpdir } from 'node:os';
import { parsePinSpec, pinCmd } from '../src/commands/pin.js';
import { readSentinel, writeSentinel } from '../src/state/sentinel.js';

let TMP: string;

before(async () => {
  TMP = await mkdtemp(join(tmpdir(), 'al-pin-'));
  process.env.AGENTLINUX_STATE_DIR = join(TMP, 'state/installed.d');
  process.env.AGENTLINUX_CATALOG_DIR = join(TMP, 'catalog');
  await mkdir(join(TMP, 'catalog'), { recursive: true });
  await writeFile(join(TMP, 'catalog/catalog.json'), JSON.stringify({
    version: '0.3.0',
    agents: [
      { id: 'foo', display_name: 'Foo', description: '.',
        source_kind: 'npm', npm_package_name: 'foo', pinned_version: '1.0.0',
        install_recipe_path: 'install.sh', uninstall_recipe_path: 'uninstall.sh' },
      { id: 'bar', display_name: 'Bar', description: '.',
        source_kind: 'script', pinned_version: '2.0.0',
        install_recipe_path: 'install.sh', uninstall_recipe_path: 'uninstall.sh' },
    ],
  }));
});
after(async () => { await rm(TMP, { recursive: true, force: true }); });

const installed = async (id: string, source: any, sticky: boolean, version = '1.0.0') => {
  await writeSentinel({ id, version, source, sticky, installed_at: '2026-04-19T00:00:00Z' });
};

describe('parsePinSpec', () => {
  test('curated target', () => {
    assert.deepEqual(parsePinSpec('foo=curated'), { name: 'foo', target: 'curated' });
  });
  test('latest target', () => {
    assert.deepEqual(parsePinSpec('foo=latest'), { name: 'foo', target: 'latest' });
  });
  test('exact semver', () => {
    assert.deepEqual(parsePinSpec('foo=2.1.7'), { name: 'foo', target: 'version', version: '2.1.7' });
  });
  test('pre-release semver', () => {
    const p = parsePinSpec('foo=2.1.7-beta.1');
    assert.equal(p.target, 'version');
    if (p.target === 'version') assert.equal(p.version, '2.1.7-beta.1');
  });
  test('invalid target throws with help', () => {
    assert.throws(() => parsePinSpec('foo=bogus'), /invalid target/);
  });
  test('no = throws', () => {
    assert.throws(() => parsePinSpec('nopeq'), /expected '<name>=<target>'/);
  });
  test('empty name throws', () => {
    assert.throws(() => parsePinSpec('=curated'), /<name>=<target>/);
  });
});

describe('pinCmd state mutation', () => {
  beforeEach(async () => {
    await rm(process.env.AGENTLINUX_STATE_DIR!, { recursive: true, force: true }).catch(() => {});
  });
  test('pin <installed>=latest: sets sticky=true + source=latest', async () => {
    await installed('foo', 'curated', false);
    await pinCmd('foo=latest');
    const s = await readSentinel('foo');
    assert.equal(s?.sticky, true);
    assert.equal(s?.source, 'latest');
    assert.equal(s?.version, '1.0.0'); // version preserved until next upgrade
  });
  test('pin <installed>=2.0.0: sets sticky=true + source=pinned + version=2.0.0', async () => {
    await installed('foo', 'curated', false);
    await pinCmd('foo=2.0.0');
    const s = await readSentinel('foo');
    assert.equal(s?.sticky, true);
    assert.equal(s?.source, 'pinned');
    assert.equal(s?.version, '2.0.0');
  });
  test('pin <installed>=curated: clears sticky + source=curated', async () => {
    await installed('foo', 'latest', true);
    await pinCmd('foo=curated');
    const s = await readSentinel('foo');
    assert.equal(s?.sticky, false);
    assert.equal(s?.source, 'curated');
  });
});
```
