    **File 1: `plugin/lib/log.sh`** (40-60 lines)

    Exact shape:
    ```bash
    #!/usr/bin/env bash
    # plugin/lib/log.sh — structured logging primitives.
    # Source-once guard: safe to `. log.sh` multiple times across library files.
    [[ -n "${AGENTLINUX_LOG_SH_SOURCED:-}" ]] && return 0
    readonly AGENTLINUX_LOG_SH_SOURCED=1

    # AGENTLINUX_LOG_LEVEL ∈ {DEBUG, INFO, WARN, ERROR}; default INFO.
    : "${AGENTLINUX_LOG_LEVEL:=INFO}"

    __log_ts() { date -u +%Y-%m-%dT%H:%M:%SZ; }

    __log_color() {
      # Echo ANSI color codes only when target FD is a tty.
      # $1 = FD number (1 or 2); $2 = color name (red|yellow|dim|reset)
      local fd=$1 color=$2
      [[ -t $fd ]] || { printf ''; return; }
      case "$color" in
        red)    printf '\033[31m' ;;
        yellow) printf '\033[33m' ;;
        dim)    printf '\033[2m'  ;;
        reset)  printf '\033[0m'  ;;
      esac
    }

    log_info()  { printf '[%s] [INFO]  %s\n' "$(__log_ts)" "$*"; }
    log_warn()  { printf '%s[%s] [WARN]  %s%s\n' "$(__log_color 2 yellow)" "$(__log_ts)" "$*" "$(__log_color 2 reset)" >&2; }
    log_error() { printf '%s[%s] [ERROR] %s%s\n' "$(__log_color 2 red)"    "$(__log_ts)" "$*" "$(__log_color 2 reset)" >&2; }
    log_debug() {
      [[ ${AGENTLINUX_LOG_LEVEL} == DEBUG ]] || return 0
      printf '%s[%s] [DEBUG] %s%s\n' "$(__log_color 2 dim)" "$(__log_ts)" "$*" "$(__log_color 2 reset)" >&2
    }
    ```

    Requirements:
    - NO external tool dependencies beyond `date`, `printf`, bash 5.x built-ins.
    - Timestamps are UTC ISO-8601 (`date -u +%Y-%m-%dT%H:%M:%SZ`) so log order across processes is total-ordered for INST-05 grep.
    - log_info → stdout (so `exec > >(tee) 2>&1` at entrypoint captures it in /var/log/agentlinux-install.log).
    - log_warn / log_error / log_debug → stderr (also captured by the tee merge at entrypoint, but separately visible during shellcheck).
    - NO `echo` — every primitive uses `printf` (shellcheck SC2028 avoidance).
    - Color functions gate on `[[ -t 2 ]]` check; log transcript in `/var/log/agentlinux-install.log` is then plain ASCII (INST-05 greps for `EACCES|permission denied` without stripping ANSI).

    ```bash
    #!/usr/bin/env bash
    # plugin/lib/as_user.sh — keystone: route every agent-owned command through sudo -u -H -E --.
    # Violating this function is the "sudo npm install -g" anti-pattern AgentLinux exists to eliminate.
    [[ -n "${AGENTLINUX_AS_USER_SH_SOURCED:-}" ]] && return 0
    readonly AGENTLINUX_AS_USER_SH_SOURCED=1

    if ! command -v log_error >/dev/null 2>&1; then

      return 1 2>/dev/null || exit 1
    fi

    # as_user <user> <cmd...> — run <cmd...> as <user> with HOME set, env preserved except secure_path.
    # `-H` forces HOME=target-user-home (load-bearing for ~/.npmrc lookups in Phase 3).

    # `--` separates sudo options from user's command.

      local user=$1
      shift

        log_error "as_user: no command given (usage: as_user <user> <cmd...>)"
        return 64
      fi
      sudo -u "$user" -H -E -- "$@"

    as_user_login() {
      local user=$1
      shift
      if [[ $# -eq 0 ]]; then
        log_error "as_user_login: no command given"
        return 64
      fi
      sudo -u "$user" -H -i -- "$@"
    }
    ```

    **File 2: `plugin/lib/idempotency.sh`** (80-120 lines)

    Exact shape:
    ```bash
    #!/usr/bin/env bash
    # plugin/lib/idempotency.sh — grep-before-mutate primitives.
    # Every state change in the installer goes through one of these helpers.
    # Blind `echo >> file` breaks INST-02 (re-run must converge).
    [[ -n "${AGENTLINUX_IDEMPOTENCY_SH_SOURCED:-}" ]] && return 0
    readonly AGENTLINUX_IDEMPOTENCY_SH_SOURCED=1

    if ! command -v log_error >/dev/null 2>&1; then
      printf 'idempotency.sh: log.sh must be sourced first\n' >&2
      return 1 2>/dev/null || exit 1
    fi

    # ensure_line_in_file <line> <file>
    # Append <line> to <file> only if not already present (fixed-string whole-line match).
    ensure_line_in_file() {
      local line=$1 file=$2
      if ! grep -Fxq -- "$line" "$file" 2>/dev/null; then
        printf '%s\n' "$line" >>"$file"
        log_info "appended line to ${file}"
      fi
    }

    # ensure_marker_block <file> <tag> [--top|--bottom]
    # Replace content between `# >>> <tag> begin >>>` / `# <<< <tag> end <<<` markers.
    # Block content on stdin. Default placement: --bottom.
    # --top is required for /home/agent/.bashrc (see RESEARCH Pitfall 2: the Ubuntu
    # default skel /.bashrc has `case $- in *i*) ;; *) return;; esac` early-return;
    # agentlinux block must precede it for `sudo -u agent bash -c` (non-interactive).
    ensure_marker_block() {
      local file=$1 tag=$2 placement=${3:---bottom}
      local begin="# >>> ${tag} begin >>>"
      local end="# <<< ${tag} end <<<"
      local content tmp
      content=$(cat)
      tmp=$(mktemp)
      trap 'rm -f "$tmp"' RETURN

      case "$placement" in
        --top)
          {
            printf '%s\n' "$begin"
            printf '%s\n' "$content"
            printf '%s\n' "$end"
            if [[ -f $file ]]; then
              awk -v b="$begin" -v e="$end" '
                $0 == b { in_block=1; next }
                $0 == e { in_block=0; next }
                !in_block { print }
              ' "$file"
            fi
          } >"$tmp"
          ;;
        --bottom)
          if [[ -f $file ]]; then
            awk -v b="$begin" -v e="$end" '
              $0 == b { in_block=1; next }
              $0 == e { in_block=0; next }
              !in_block { print }
            ' "$file" >"$tmp"
          fi
          {
            printf '%s\n' "$begin"
            printf '%s\n' "$content"
            printf '%s\n' "$end"
          } >>"$tmp"
          ;;
        *)
          log_error "ensure_marker_block: unknown placement '${placement}' (expected --top or --bottom)"
          return 64
          ;;
      esac

      install -m 0644 "$tmp" "$file"
      log_info "wrote marker block '${tag}' (${placement}) to ${file}"
    }

    # ensure_user <name>
    # useradd only if absent. Creates home, bash shell, matching user-group.
    ensure_user() {
      local user=$1
      if id "$user" >/dev/null 2>&1; then
        log_info "user ${user} already exists (no-op)"
        return 0
      fi
      useradd --create-home --shell /bin/bash --user-group "$user"
      log_info "created user ${user} (home /home/${user}, shell /bin/bash)"
    }

    # ensure_dir <path> <mode> <user:group>
    # Create directory if absent; enforce mode+ownership unconditionally.
    ensure_dir() {
      local path=$1 mode=$2 owner=$3
      if [[ ! -d $path ]]; then
        install -d -m "$mode" -o "${owner%:*}" -g "${owner#*:}" "$path"
        log_info "created directory ${path} (${mode} ${owner})"
      else
        chmod "$mode" "$path"
        chown "$owner" "$path"
      fi
    }

    # visudo_validate <file>
    # `visudo -cf <file>` — safety check before installing a sudoers drop-in.
    # Phase 2 ships no drop-ins; this helper exists so Phase 3+ can't forget it.
    visudo_validate() {
      local file=$1
      if ! visudo -cf "$file" >/dev/null; then
        log_error "sudoers syntax check failed for ${file} (visudo -cf rejected)"
        return 1
      fi
      log_info "sudoers syntax OK: ${file}"
    }
    ```

    Requirements:
    - `ensure_line_in_file` uses `grep -Fxq --` (fixed string, whole-line match, -- to stop flag parsing on content that starts with `-`).
    - `ensure_marker_block` cleans up tmp file via `trap 'rm -f "$tmp"' RETURN` (bash function-scoped trap).
    - `ensure_marker_block --top` MUST place block before any existing content — see the awk-after-printf ordering in the --top branch.
    - `visudo_validate` exists even though Phase 2 writes no drop-in — Phase 3+ callers cannot bypass it.
    - NO raw `echo >> file` anywhere.
    - NO `sed -i` (edge cases with newlines, GNU vs BSD, atomicity).

    Then format: `shfmt -i 2 -ci -bn -w plugin/lib/as_user.sh plugin/lib/idempotency.sh`
