```typescript
// plugin/cli/src/runner.ts — shared recipe dispatcher for install/remove/upgrade.
// Mirrors /etc/agentlinux.env (plugin/provisioner/40-path-wiring.sh) byte-for-byte
// so recipes inherit the same PATH/locale whether triggered by provisioner or CLI.

import { asUser } from './state/dispatcher.js';
import type { CatalogEntry } from './types.js';

export interface RecipeEnv {
  AGENTLINUX_PINNED_VERSION: string;
  AGENTLINUX_CATALOG_DIR: string;
  AGENTLINUX_AGENT_HOME: string;
  AGENTLINUX_SOURCE_KIND: string;
  AGENTLINUX_INSTALL_LOG: string;
  [key: string]: string;
}

// Canonical PATH literal — MUST be byte-identical to /etc/agentlinux.env and
// /etc/cron.d/agentlinux written by 40-path-wiring.sh. T-04-07 mitigation:
// a drift between these two would cause recipes to resolve npm from /usr/bin
// (Pitfall 5) and break the no-EACCES contract.
export const AGENT_PATH =
  '/home/agent/.npm-global/bin:/home/agent/.local/bin:/usr/local/bin:/usr/bin:/bin';

export interface DispatchArgs {
  entry: CatalogEntry;
  recipePath: string;       // absolute path to install.sh / uninstall.sh
  version: string;          // decideVersion().version
  catalogDir: string;       // Catalog.catalogDir
  extraEnv?: Record<string, string>;
}

export async function dispatchRecipe(
  args: DispatchArgs
): Promise<{ exitCode: number; stdout: string; stderr: string }> {
  const env: Record<string, string> = {
    AGENTLINUX_PINNED_VERSION: args.version,
    AGENTLINUX_CATALOG_DIR: args.catalogDir,
    AGENTLINUX_AGENT_HOME: '/home/agent',
    AGENTLINUX_SOURCE_KIND: args.entry.source_kind,
    AGENTLINUX_INSTALL_LOG: '/var/log/agentlinux-install.log',
    PATH: AGENT_PATH,
    HOME: '/home/agent',
    NPM_CONFIG_PREFIX: '/home/agent/.npm-global',
    LANG: 'C.UTF-8',
    LC_ALL: 'C.UTF-8',
    ...(args.extraEnv ?? {}),
  };

  return asUser('agent', ['bash', args.recipePath], { env });
}
```

```typescript
import { loadCatalog } from '../catalog/loader.js';
import { listSentinels, readSentinel } from '../state/sentinel.js';
import { classify } from '../version/classify.js';
import type { CatalogEntry, Sentinel, Status } from '../types.js';

export interface ListOpts { json?: boolean; includeTest?: boolean; }

interface Row {
  id: string;
  display_name: string;
  status: Status;
  curated: string;          // entry.pinned_version
  installed: string;        // sentinel?.version ?? '-'
  description: string;
  source: Sentinel['source'] | '-';
}

export async function listCmd(opts: ListOpts): Promise<void> {
  // Hot path: skip ajv validation (RESEARCH Open Q2 + Pitfall 2 resolution).
  // Pre-commit + install/upgrade paths catch schema errors early; list
  // tolerates a partially-invalid catalog so it can still show info.
  const catalog = await loadCatalog({ validate: false });
  const sentinels = await listSentinels();
  const bySentinel = new Map(sentinels.map((s) => [s.id, s]));

  const visible = catalog.agents.filter((a) => opts.includeTest || !a.test_only);

  const rows: Row[] = visible.map((entry) => {
    const sentinel = bySentinel.get(entry.id) ?? null;
    // For Phase 4, `installed` is the sentinel's recorded version.
    // Phase 5 AGT-XX may cross-check against `npm ls -g --json` for drift
    // detection; for now, sentinel-version is the truth we record.
    const installed = sentinel?.version ?? null;
    const status = classify({ entry, sentinel, installed });
    return {
      id: entry.id,
      display_name: entry.display_name,
      status,
      curated: entry.pinned_version,
      installed: installed ?? '-',
      description: entry.description,
      source: sentinel?.source ?? '-',
    };
  });

  if (opts.json) {
    console.log(JSON.stringify(rows, null, 2));
    return;
  }

  // Text table: NAME STATUS CURATED INSTALLED DESCRIPTION
  // Grep-friendly — no color, no box chars.
  const header = ['NAME', 'STATUS', 'CURATED', 'INSTALLED', 'DESCRIPTION'];
  const data = rows.map((r) => [r.id, r.status, r.curated, r.installed, r.description]);
  const all = [header, ...data];
  const widths = header.map((_, i) => Math.max(...all.map((row) => row[i].length)));
  for (const row of all) {
    console.log(row.map((c, i) => c.padEnd(widths[i])).join('  '));
  }
}
```

```typescript
import { join } from 'node:path';
import semver from 'semver';
import { loadCatalog } from '../catalog/loader.js';
import { readSentinel, writeSentinel } from '../state/sentinel.js';
import { decideVersion } from '../version/classify.js';
import { dispatchRecipe } from '../runner.js';

export interface InstallOpts { force?: boolean; version?: string; json?: boolean; includeTest?: boolean; }

export async function installCmd(name: string, opts: InstallOpts): Promise<void> {
  const catalog = await loadCatalog({ validate: true });
  const entry = catalog.agents.find((a) => a.id === name);

  if (!entry) {
    console.error(`agentlinux: no such agent in catalog: ${name}`);
    console.error(`  available: ${catalog.agents.filter((a) => !a.test_only).map((a) => a.id).join(', ')}`);
    process.exit(64);
  }

  // Honor test_only flag unless --include-test provided. RESEARCH §Pattern 11.
  if (entry.test_only && !opts.includeTest) {
    console.error(`agentlinux: ${name} is a test-only entry; pass --include-test to install`);
    process.exit(64);
  }

  if (opts.version !== undefined && !semver.valid(opts.version)) {
    console.error(`agentlinux: --version '${opts.version}' is not a valid semver`);
    process.exit(64);
  }

  const existing = await readSentinel(entry.id);
  const decision = decideVersion(entry, opts.version, existing);

  // Idempotent short-circuit: same version + not --force → no-op.
  if (!opts.force && existing && semver.eq(existing.version, decision.version)) {
    console.log(`${entry.id}: already installed at ${existing.version} (${existing.source}); no-op`);
    return;
  }

  const recipePath = join(catalog.catalogDir, 'agents', entry.id, entry.install_recipe_path);
  const result = await dispatchRecipe({
    entry, recipePath, version: decision.version, catalogDir: catalog.catalogDir,
  });

  if (result.exitCode !== 0) {
    console.error(`${entry.id}: install.sh failed (exit ${result.exitCode})`);
    if (result.stderr) console.error(result.stderr);
    process.exit(result.exitCode);
  }
  if (result.stdout) console.log(result.stdout.trimEnd());

  await writeSentinel({
    id: entry.id,
    version: decision.version,
    source: decision.source,
    sticky: decision.sticky,
    installed_at: new Date().toISOString(),
  });

  console.log(`${entry.id}: installed ${decision.version} (${decision.source})`);
}
```

```typescript
import { join } from 'node:path';
import { loadCatalog } from '../catalog/loader.js';
import { readSentinel, deleteSentinel } from '../state/sentinel.js';
import { dispatchRecipe } from '../runner.js';

export interface RemoveOpts { force?: boolean; }

export async function removeCmd(name: string, opts: RemoveOpts): Promise<void> {
  const catalog = await loadCatalog({ validate: true });
  const entry = catalog.agents.find((a) => a.id === name);
  if (!entry) {
    console.error(`agentlinux: no such agent in catalog: ${name}`);
    process.exit(64);
  }

  const sentinel = await readSentinel(entry.id);
  if (!sentinel) {
    if (!opts.force) {
      console.error(`agentlinux: ${entry.id} is not installed (pass --force for no-op)`);
      process.exit(1);
    }
    return; // --force on not-installed = no-op
  }

  const recipePath = join(catalog.catalogDir, 'agents', entry.id, entry.uninstall_recipe_path);
  const result = await dispatchRecipe({
    entry, recipePath, version: sentinel.version, catalogDir: catalog.catalogDir,
  });
  if (result.exitCode !== 0) {
    console.error(`${entry.id}: uninstall.sh failed (exit ${result.exitCode})`);
    if (result.stderr) console.error(result.stderr);
    process.exit(result.exitCode);
  }
  if (result.stdout) console.log(result.stdout.trimEnd());

  await deleteSentinel(entry.id);
  console.log(`${entry.id}: removed`);
}
```
