Step 2 — Create plugin/cli/src/upgrade/divergence.ts:
```typescript
import semver from 'semver';
import type { CatalogEntry, Sentinel, DivergenceReport } from '../types.js';
import { classify } from '../version/classify.js';

/** Pure function: builds a per-agent DivergenceReport from the 4 inputs. */
export function computeDivergence(input: {
  entry: CatalogEntry;
  sentinel: Sentinel | null;
  installed: string | null;
  latest?: string | null;
}): DivergenceReport {
  const { entry, sentinel, installed, latest = null } = input;
  const status = classify({ entry, sentinel, installed });
  return {
    id: entry.id,
    status,
    sentinelVersion: sentinel?.version ?? null,
    installedVersion: installed,
    curatedVersion: entry.pinned_version,
    latestVersion: latest,
    source: sentinel?.source ?? 'none',
    sticky: sentinel?.sticky ?? false,
  };
}

/**
 * Resolve the highest published version that satisfies entry.version_constraint.
 * Absent constraint = newest published. Used by --all-latest and --check-upstream.
 * Throws when constraint matches zero versions (RESEARCH §Pattern 7 pitfall check).
 */
export function resolveLatestFor(
  entry: CatalogEntry,
  publishedVersions: string[]
): string {
  if (publishedVersions.length === 0) {
    throw new Error(`${entry.id}: no published versions found`);
  }
  if (!entry.version_constraint) {
    // npm returns versions chronologically; use semver to find true maximum.
    const max = semver.maxSatisfying(publishedVersions, '*');
    if (!max) throw new Error(`${entry.id}: cannot resolve newest from published versions`);
    return max;
  }
  const max = semver.maxSatisfying(publishedVersions, entry.version_constraint);
  if (!max) {
    throw new Error(
      `${entry.id}: no published version of ${entry.npm_package_name ?? entry.id} ` +
      `satisfies constraint ${entry.version_constraint}`
    );
  }
  return max;
}
```

Step 3 — Create plugin/cli/src/upgrade/npm_ls.ts:
```typescript
import { asUser } from '../state/dispatcher.js';
import type { CatalogEntry } from '../types.js';
import { resolveLatestFor } from './divergence.js';
import { AGENT_PATH } from '../runner.js';

const NPM_ENV = {
  PATH: AGENT_PATH,
  HOME: '/home/agent',
  NPM_CONFIG_PREFIX: '/home/agent/.npm-global',
  LANG: 'C.UTF-8',
  LC_ALL: 'C.UTF-8',
};

/**
 * Query the agent user's globally-installed npm packages.
 * Returns Map<pkg_name, version>. Missing-peer-dep warnings cause npm ls to
 * exit 1 but still emit valid JSON — Pitfall 4: parse stdout regardless of
 * exit code, fail only on unparseable JSON.
 */
export async function queryGlobalNpm(
  dispatcher: typeof asUser = asUser
): Promise<Map<string, string>> {
  const result = await dispatcher('agent', ['npm', 'ls', '-g', '--json', '--depth=0'], { env: NPM_ENV });
  // npm ls may exit 1 on peer-dep warnings; JSON is still emitted. Only fail
  // if stdout doesn't parse (genuine npm misbehavior).
  let parsed: { dependencies?: Record<string, { version?: string; overridden?: boolean }> };
  try {
    parsed = JSON.parse(result.stdout);
  } catch (err) {
    throw new Error(
      `npm ls -g --json failed to emit parseable JSON (exit ${result.exitCode}):\n${result.stderr}`
    );
  }
  const map = new Map<string, string>();
  for (const [pkg, info] of Object.entries(parsed.dependencies ?? {})) {
    if (info?.version) map.set(pkg, info.version);
  }
  return map;
}

/**
 * Query npm view to resolve the upstream-latest version for an agent.
 * Honors entry.version_constraint via semver.maxSatisfying. Returns null if
 * the entry is not npm-backed (source_kind !== 'npm').
 * Only called when user opts in via --check-upstream / --all-latest.
 */
export async function queryNpmViewLatest(
  entry: CatalogEntry,
  dispatcher: typeof asUser = asUser
): Promise<string | null> {
  if (entry.source_kind !== 'npm' || !entry.npm_package_name) {
    // Native installers (claude-code script) don't have a single canonical
    // "latest" we can derive from npm — return null and the upgrade flow
    // treats latestVersion as unknown.
    return null;
  }
  const result = await dispatcher(
    'agent',
    ['npm', 'view', entry.npm_package_name, 'versions', '--json'],
    { env: NPM_ENV, timeout: 30_000 }
  );
  if (result.exitCode !== 0) {
    throw new Error(
      `npm view ${entry.npm_package_name} failed (exit ${result.exitCode}):\n${result.stderr}`
    );
  }
  const versions: string[] = JSON.parse(result.stdout);
  return resolveLatestFor(entry, versions);
}
```

```typescript
import { loadCatalog } from '../catalog/loader.js';
import { listSentinels, readSentinel, writeSentinel } from '../state/sentinel.js';
import { computeDivergence } from '../upgrade/divergence.js';
import { queryGlobalNpm, queryNpmViewLatest } from '../upgrade/npm_ls.js';
import { dispatchRecipe } from '../runner.js';
import type { CatalogEntry, DivergenceReport, Sentinel } from '../types.js';

export interface UpgradeOpts {
  resetAllCurated?: boolean;
  respectOverrides?: boolean;
  allLatest?: boolean;
  checkUpstream?: boolean;
  json?: boolean;
}

type Dispatcher = typeof dispatchRecipe;

function willTouchUpstream(opts: UpgradeOpts): boolean {
  return opts.checkUpstream === true || opts.allLatest === true;
}

function shouldReinstall(report: DivergenceReport, opts: UpgradeOpts): 'curated' | 'latest' | null {
  if (opts.resetAllCurated) return 'curated';
  if (opts.allLatest && !report.sticky) return 'latest';
  if (opts.respectOverrides) {
    if (report.source === 'curated' && report.status !== 'synced') return 'curated';
    return null;
  }
  return null;
}

export async function upgradeCmd(
  opts: UpgradeOpts,
  deps: {
    dispatchRecipe?: Dispatcher;
  } = {}
): Promise<void> {
  const dispatch = deps.dispatchRecipe ?? dispatchRecipe;
  const catalog = await loadCatalog({ validate: true });
  const sentinels = await listSentinels();
  const bySentinel = new Map(sentinels.map((s) => [s.id, s]));
  const npmLs = await queryGlobalNpm();

  const reports: DivergenceReport[] = [];
  for (const entry of catalog.agents) {
    const sentinel = bySentinel.get(entry.id) ?? null;
    // installed version: npm ls for npm entries; for script/native entries
    // (claude-code) we defer to sentinel.version (Phase 5 may add a native
    // version-probe; for now, Phase 4 scope is npm-driven drift detection).
    const installed = entry.source_kind === 'npm' && entry.npm_package_name
      ? npmLs.get(entry.npm_package_name) ?? null
      : sentinel?.version ?? null;
    let latest: string | null = null;
    if (willTouchUpstream(opts) && entry.source_kind === 'npm') {
      try {
        latest = await queryNpmViewLatest(entry);
      } catch (err: any) {
        console.error(`  ! ${entry.id}: could not resolve latest — ${err.message}`);
      }
    }
    reports.push(computeDivergence({ entry, sentinel, installed, latest }));
  }

  if (opts.json) {
    console.log(JSON.stringify(reports, null, 2));
  } else {
    // Grep-friendly table. No color.
    const header = ['ID', 'STATUS', 'SENTINEL', 'INSTALLED', 'CURATED', 'LATEST', 'SRC'];
    const rows = reports.map((r) => [
      r.id, r.status,
      r.sentinelVersion ?? '-', r.installedVersion ?? '-',
      r.curatedVersion, r.latestVersion ?? '-', r.source,
    ]);
    const all = [header, ...rows];
    const widths = header.map((_, i) => Math.max(...all.map((row) => row[i].length)));
    for (const row of all) {
      console.log(row.map((c, i) => c.padEnd(widths[i])).join('  '));
    }
  }

  // No bulk flag → report-only.
  const isReportOnly = !opts.resetAllCurated && !opts.respectOverrides && !opts.allLatest;
  if (isReportOnly) return;

  // Reconcile loop.
  const entryById = new Map(catalog.agents.map((e) => [e.id, e]));
  for (const report of reports) {
    const target = shouldReinstall(report, opts);
    if (!target) continue;
    const entry = entryById.get(report.id)!;
    let version: string;
    let source: Sentinel['source'];
    if (target === 'latest') {
      if (!report.latestVersion) {
        console.error(`${entry.id}: skipping (no upstream latest resolved)`);
        continue;
      }
      version = report.latestVersion;
      source = 'latest';
    } else {
      version = entry.pinned_version;
      source = 'curated';
    }
    const recipePath = join(catalog.catalogDir, 'agents', entry.id, entry.install_recipe_path);
    console.log(`${entry.id}: reinstalling at ${version} (${source})`);
    const result = await dispatch({ entry, recipePath, version, catalogDir: catalog.catalogDir });
    if (result.exitCode !== 0) {
      console.error(`${entry.id}: recipe failed (exit ${result.exitCode})`);
      if (result.stderr) console.error(result.stderr);
      continue;
    }
    if (result.stdout) console.log(result.stdout.trimEnd());
    await writeSentinel({
      id: entry.id, version, source,
      sticky: source === 'latest' ? (await readSentinel(entry.id))?.sticky ?? false : false,
      installed_at: new Date().toISOString(),
    });
  }
}
```
